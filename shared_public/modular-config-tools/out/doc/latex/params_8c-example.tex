\textcolor{keyword}{static} \textcolor{keywordtype}{void} PrintCommandOptions(\textcolor{keywordtype}{void})
\{
  printf(\textcolor{stringliteral}{"Application options:\(\backslash\)n"});
  \textcolor{keywordtype}{size\_t} i = 0U;
  \textcolor{keywordflow}{while}(arstCommandlineOptions[i].name != NULL)
  \{
    \textcolor{keywordtype}{char} \textcolor{keyword}{const} * szParam = \textcolor{stringliteral}{""};
    \textcolor{keywordflow}{if}(arstCommandlineOptions[i].has\_arg == required\_argument)
    \{
      szParam = \textcolor{stringliteral}{"<param>"};
    \}
    \textcolor{keywordflow}{if}(arstCommandlineOptions[i].has\_arg == optional\_argument)
    \{
      szParam = \textcolor{stringliteral}{"[param]"};
    \}
    printf(\textcolor{stringliteral}{"  --%-25s %-7s   -%c   %s\(\backslash\)n"}, arstCommandlineOptions[i].name,
                                          szParam,
                                          arstCommandlineOptions[i].val,
                                          arszCommandlineOptionTexts[i]);
    i++;
  \}

  \textcolor{keywordflow}{return};
\}
